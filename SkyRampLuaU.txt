local function updateRoot()

        if rampFolder then
            firstPart = rampFolder:FindFirstChild("firstPart")
            lastPart = rampFolder:FindFirstChild("LastPartBruh")
            Pplatform = rampFolder:FindFirstChild("Platform")

            if firstPart and Pplatform then
                Pplatform.CFrame = firstPart.CFrame * CFrame.new(0, Pplatform.Size.Y/2 + 2, 0)
            end
        end
    end
end)

local function createButton(text, posY, color)
    local btn = Instance.new("TextButton")
    btn.Parent = ui
    btn.Transparency = 0.2
    btn.BackgroundColor3 = Color3.new(0, 0, 0)
    btn.Size = UDim2.new(0.1, 0, 0.02, 0)
    btn.Position = UDim2.new(0.007, 0, posY, 0)
    btn.Font = Enum.Font.BuilderSansExtraBold
    btn.Text = text
    btn.TextScaled = true
    local stroke = Instance.new("UIStroke")
    stroke.Color = color
    stroke.Parent = btn
    stroke.Thickness = 2
    local stroke2 = stroke:Clone()
	stroke2.Name = "UIStroke2"
    stroke2.Parent = btn
    stroke2.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
    return btn
end

SpawnButton.MouseButton1Click:Connect(function()
    if hasSpawnedRamp or not root then return end
    hasSpawnedRamp = true
    lastPart = nil
    firstPart = nil

    rampFolder = Instance.new("Folder")
    rampFolder.Name = "TeachersSkyRamp"
    rampFolder.Parent = workspace

    for i = 1, 89 do
        local part = Instance.new("Part")
        part.BrickColor = BrickColor.new("Cashmere")
        part.Material = Enum.Material.WoodPlanks
        part.Parent = rampFolder
        part.Anchored = true
        part.CastShadow = false

        if i == 1 then
            firstPart = part
            part.Name = "firstPart"
        end

        local part2 = Instance.new("Part")
        part2.BrickColor = BrickColor.new("Cashmere")
        part2.Material = Enum.Material.WoodPlanks
        part2.Parent = rampFolder
        part2.Anchored = true
        part2.Size = Vector3.new(5,15,20)

        local part3 = Instance.new("Part")
        part3.BrickColor = BrickColor.new("Cashmere")
        part3.Material = Enum.Material.WoodPlanks
        part3.Parent = rampFolder
        part3.Anchored = true
        part3.Size = Vector3.new(5,15,20)

        if lastPart then
            part2.CFrame = part.CFrame * CFrame.new(part.Size.X/2 + part2.Size.X/2, part2.Size.Y/2, 0)
            part3.CFrame = part.CFrame * CFrame.new(-(part.Size.X/2 + part3.Size.X/2), part3.Size.Y/2, 0)
        end

        if i == 35 or i == 36 or i == 37 or i == 60 then
            part2.Size = Vector3.new(5,15,2500)
            part3.Size = Vector3.new(5,15,2500)
            part.Size = Vector3.new(100,2,2500)
            if lastPart then
                part.CFrame = lastPart.CFrame * CFrame.new(0, part.Size.Y/2, -(lastPart.Size.Z/2 + part.Size.Z/500)) * CFrame.Angles(math.rad(1),0,0)
                part2.CFrame = part.CFrame * CFrame.new(part.Size.X/2 + part2.Size.X/2, part2.Size.Y/2, 0)
                part3.CFrame = part.CFrame * CFrame.new(-(part.Size.X/2 + part3.Size.X/2), part3.Size.Y/2, 0)
            end
        else
            part2.Size = Vector3.new(5,15,20)
            part3.Size = Vector3.new(5,15,20)
            part.Size = Vector3.new(100,0.2,20)
        end

        if i == 1 then
            part.CFrame = root.CFrame * CFrame.new(0,-4,-1)
        elseif i ~= 35 and i ~=36 and i ~=37 and i ~=60 then
            part.CFrame = lastPart.CFrame * CFrame.new(0, part.Size.Y/2, -(lastPart.Size.Z/2.5 + part.Size.Z/2.5)) * CFrame.Angles(math.rad(1),0,0)
            part2.CFrame = part.CFrame * CFrame.new(part.Size.X/2 + part2.Size.X/2, part2.Size.Y/2, 0)
            part3.CFrame = part.CFrame * CFrame.new(-(part.Size.X/2 + part3.Size.X/2), part3.Size.Y/2, 0)
        end

        if i == 89 then part.Name = "LastPartBruh" end
        lastPart = part
    end

	local platform = Instance.new("Part")
    platform.Name = "platform"
    platform.Parent = rampFolder
    platform.Size = Vector3.new(100,250,20)
    platform.Anchored = true
    platform.BrickColor = BrickColor.new("Cashmere")
    platform.Material = Enum.Material.WoodPlanks
    platform.CFrame = lastPart.CFrame * CFrame.new(0,-125,0)
    pPlatform = platform

    for i = 1, 6 do
        local part = Instance.new("Part")
        part.Parent = rampFolder
        part.Anchored = true
        part.Size = Vector3.new(100, 5.2, 30)
        part.BrickColor = BrickColor.new("Cashmere")
        part.Material = Enum.Material.WoodPlanks

        if i == 1 then
            part.CFrame = root.CFrame * CFrame.new(0, -6, 10) * CFrame.Angles(0, math.rad(180), 0)
        else
            part.CFrame = lastPart.CFrame * CFrame.new(0, part.Size.Y/2, -(lastPart.Size.Z/2 + part.Size.Z/2)) * CFrame.Angles(math.rad(10), 0, 0)
        end

        lastPart = part
    end

    local Platform = Instance.new("Part")
    Platform.Name = "Platform"
    Platform.Parent = rampFolder
    Platform.Size = Vector3.new(100,50,2500)
    Platform.Anchored = true
    Platform.CFrame = root.CFrame - Vector3.new(0,60,0)
    Pplatform = Platform
end)

callButton.MouseButton1Click:Connect(function()
    if isCalling then return end
    if not hasSpawnedRamp or not rampFolder then return end
    isCalling = true

    local Platform = rampFolder:FindFirstChild("Platform")
    local lastPart = rampFolder:FindFirstChild("LastPartBruh")
    if not Platform or not lastPart then isCalling = false return end

    local startPos = Platform.Position
    local yAlignPos = Vector3.new(startPos.X, lastPart.Position.Y + Platform.Size.Y/2, startPos.Z)
    local fullPos = lastPart.Position + Vector3.new(0, Platform.Size.Y/2, 0)
    local resetPos = root.Position - Vector3.new(0,60,0)

    local tween1 = tweenService:Create(Platform, TweenInfo.new(20, Enum.EasingStyle.Linear), {Position = yAlignPos})
    tween1:Play()
    tween1.Completed:Wait()

    wait(1)

    local tween2 = tweenService:Create(Platform, TweenInfo.new(20, Enum.EasingStyle.Linear), {Position = fullPos})
    tween2:Play()
    tween2.Completed:Wait()

    wait(5)

    local tween3 = tweenService:Create(Platform, TweenInfo.new(1.5, Enum.EasingStyle.Linear), {Position = resetPos})
    tween3:Play()
    tween3.Completed:Wait()

    isCalling = false
end)

DeleteButton.MouseButton1Click:Connect(function()
    hasSpawnedRamp = false
    if rampFolder then
        for _, thing in pairs(rampFolder:GetChildren()) do
            thing:Destroy()
        end
        rampFolder:Destroy()
        rampFolder = nil
    end
    isCalling = false
end)
